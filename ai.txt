<script>
  const messagesEl = document.getElementById('messages');
  const userInput = document.getElementById('user-input');
  const sendBtn = document.getElementById('send-btn');

  let classLevel = "";
  let subject = "";

  function append(msg, cls) {
    const d = document.createElement('div');
    d.className = `message ${cls}`;
    d.innerText = msg;
    messagesEl.appendChild(d);
    messagesEl.scrollTop = messagesEl.scrollHeight;
  }

  function askOptions(question, options, handler) {
    append(question, 'bot');
    const container = document.createElement('div');
    container.className = 'options';
    options.forEach(opt => {
      const b = document.createElement('button');
      b.className = 'option-btn';
      b.innerText = opt;
      b.onclick = () => {
        append(opt, 'user');
        container.remove();
        handler(opt);
      };
      container.appendChild(b);
    });
    messagesEl.appendChild(container);
    messagesEl.scrollTop = messagesEl.scrollHeight;
  }

  function start() {
    askOptions("Welcome! Select your class:", ["Class 9","Class 10"], cls => {
      classLevel = cls;
      askOptions("Select subject:", ["Math","Physics","Chemistry"], sub => {
        subject = sub;
        append(`You chose ${classLevel} - ${subject}. Ask your question!`, 'bot');
      });
    });
  }

  function appendTyping() {
    append("Typing...", 'bot typing');
  }
  function removeTyping() {
    const t = messagesEl.querySelector('.typing');
    if (t) t.remove();
  }

  async function askPuter(question) {
    appendTyping();
    try {
      const resp = await puter.ai.chat(question, {
        model: 'gpt-4.1-nano'
      });
      removeTyping();
      append(resp, 'bot');
    } catch {
      removeTyping();
      append("Error contacting AI.", 'bot');
    }
  }

  sendBtn.onclick = () => {
    const txt = userInput.value.trim();
    if (!txt) return;
    append(txt, 'user');
    userInput.value = '';

    // Only ask AI after initial selections
    if (classLevel && subject) {
      // Add context
      askPuter(`[Class ${classLevel} ${subject} Student] ${txt}`);
    } else {
      append("Please select class and subject first.", 'bot');
    }
  };

  userInput.addEventListener('keypress', (e) => {
    if (e.key === 'Enter') sendBtn.click();
  });

  // Start the flow
  start();
</script>
